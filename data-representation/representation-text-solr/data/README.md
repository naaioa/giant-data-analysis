

目录说明:

内嵌测试用的Solr目录.

`solr.xml`, `solrschema.xml`, `schema.xml`以及`core.properties`参照`SOLR_HOME/server/solr/configsets/basic_configs`目录下配置.


	data
	├── README.md
	└── solr
	    ├── book
	    │   ├── conf
	    │   │   ├── currency.xml
	    │   │   ├── lang
	    │   │   │   └── stopwords_en.txt
	    │   │   ├── protwords.txt
	    │   │   ├── schema.xml
	    │   │   ├── solrconfig.xml
	    │   │   ├── stopwords.txt
	    │   │   └── synonyms.txt
	    │   ├── core.properties
	    │   └── data
	    │       ├── _readme		// 自定义占位符
	    │       ├── index			// 索引存放目录
	    │       └── tlog			// 日志存放目录
	    └── solr.xml
